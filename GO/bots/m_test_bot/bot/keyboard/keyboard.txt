func CreateMenu() *tele.ReplyMarkup {
	products := tele.InlineButton{
		Unique: "products_text",
		Text:   "Продукты",
	}
	special_offer := tele.InlineButton{
		Unique: "so_text",
		Text:   "Акции",
	}
	orders := tele.InlineButton{
		Unique: "orders_text",
		Text:   "Заказ",
	}
	contacts := tele.InlineButton{
		Unique: "contacts_text",
		Text:   "Контакты",
	}
	delevery := tele.InlineButton{
		Unique: "delivery_text",
		Text:   "Доставка",
	}
	info := tele.InlineButton{
		Unique: "info",
		Text:   "Инфо о пользователе ",
	}

	return &tele.ReplyMarkup{
		InlineKeyboard: [][]tele.InlineButton{
			{products},
			{special_offer, orders},
			{delevery, contacts},
			{info},
		},
	}
}
